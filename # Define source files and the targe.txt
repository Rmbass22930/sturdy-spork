# Define source files and the target EXE location
$sourceDir = "G:\BallisticTarget"  # This assumes the program is currently on G:
$exeFile = "G:\BallisticTarget\dist\BallisticTarget.exe"

# Check if the EXE exists in the source directory
if (-not (Test-Path $exeFile)) {
    Write-Host "ERROR: The EXE file was not found at $exeFile" -ForegroundColor Red
    exit
}

# Ask user for the USB drive letter where the program should be installed
$usbDriveLetter = Read-Host "Enter the drive letter of your USB drive (e.g., I, J)"
$usbDrivePath = "$usbDriveLetter`:\"

# Create target folders on USB if not already there
$usbOutputDir = "$usbDrivePath\output\targets"
$usbLogsDir = "$usbDrivePath\logs"
$usbConfigPath = "$usbDrivePath\config.json"

Write-Host "Creating target directories on USB..." -ForegroundColor Cyan

# Make sure the folders exist, create if not
New-Item -ItemType Directory -Force -Path $usbOutputDir
New-Item -ItemType Directory -Force -Path $usbLogsDir

# Copy the EXE and necessary files to the USB
Write-Host "Copying EXE and necessary files to USB..." -ForegroundColor Cyan
Copy-Item "$exeFile" -Destination "$usbDrivePath"
Copy-Item "G:\BallisticTarget\config.json" -Destination "$usbConfigPath" -Force

# Create a PowerShell script on the USB to run the EXE
$runScriptPath = "$usbDrivePath\Run_BallisticTarget.ps1"
$runScriptContent = @"
# Run_BallisticTarget.ps1
# This script will launch BallisticTarget from the USB drive.

$ErrorActionPreference = "Stop"

# Set the USB directory path dynamically
\$usbRoot = "\$PSScriptRoot"

# Define the EXE path
\$exePath = "\$usbRoot\BallisticTarget.exe"

# Check if the EXE exists on the USB drive
if (-not (Test-Path \$exePath)) {
    Write-Host "ERROR: BallisticTarget.exe not found on USB drive." -ForegroundColor Red
    exit
}

# Launch the EXE
Start-Process -FilePath \$exePath
Write-Host "BallisticTarget launched from USB." -ForegroundColor Green
"@

# Write the script content to the USB
Set-Content -Path $runScriptPath -Value $runScriptContent -Force

Write-Host "Done!" -ForegroundColor Green
Write-Host "To run the app, double-click 'Run_BallisticTarget.ps1' on the USB drive." -ForegroundColor Green
