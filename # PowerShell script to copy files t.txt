# PowerShell script to copy files to USB drive
# Ask user for the source and destination drive letters

$sourceDrive = Read-Host "Enter the source drive letter (e.g., G: for the source)"
$destinationDrive = Read-Host "Enter the destination drive letter (e.g., I: for USB drive)"

# Define source and destination paths
$sourcePath = "$sourceDrive\BallisticTarget\*"
$destinationPath = "$destinationDrive\BallisticTarget"

# Verify if source path exists
if (Test-Path $sourcePath) {
    # Create destination directory if it doesn't exist
    if (-not (Test-Path $destinationPath)) {
        New-Item -Path $destinationDrive -Name "BallisticTarget" -ItemType Directory
    }
    
    # Copy files from source to destination
    Copy-Item $sourcePath -Destination $destinationPath -Recurse -Force
    Write-Host "Files copied successfully to USB drive!"
} else {
    Write-Host "Source path does not exist. Please check the source drive and try again."
}
